var fibGenerator = function*() { // 这里的 * 表示它是生成器
    let current = 0;
    let next = 1;
 
    while (true) {
        // 1. 产出当前的值，然后函数在这里"冻结"暂停！
        // 等待外部调用 .next() 才会继续往下执行
        yield current;
     
        // 2. 重新计算数值（为了下一次苏醒做准备）
        // 利用解构赋值同时更新两个变量 (JS 的优雅写法)
        [current, next] = [next, current + next];
    }
};
 
  const gen = fibGenerator(); // 初始化，此时代码一行都没跑

let arr = [1, -2, 3, 4, -9, 6]
function getMaxSubSum(arr){
    let max = arr[0]
    let relstr = []
    let range = arr.length -1
    for(let i = 0; i <= range ;i++){
    let acc = 0
        for(let j = i ; j <= range ; j++){
            acc += arr[j]
            // max = (max > acc) ? max:acc
           if(max <= acc){
              max = acc
              relstr = arr.slice(i,j+1)
           }
        }
    }
    return relstr
}

var addTwoNumbers = function(l1, l2) {
    // 1. 造一个假的头，方便后面挂节点
    let dummy = new ListNode(0);
    let current = dummy; // current 是个搬砖工，负责一直往后铺路
    let carry = 0;       // 进位
 
    // 2. 循环条件：只要 l1 还没走完，或者 l2 还没走完，或者 还有进位没处理
    // 就可以一直循环 ( || 表示“或者”)
    while (l1 !== null || l2 !== null || carry > 0) {
     
        // === 第一步：取值 ===
        // 如果 l1 还是个节点，就取它的值；如果是 null，就当成 0
        let val1 = (l1 !== null) ? l1.val : 0;
        let val2 = (l2 !== null) ? l2.val : 0;
     
        // === 第二步：求和 ===
        // sum = val1 + val2 + 之前的进位
        let sum = val1 + val2 + carry;
     
        // === 第三步：算新进位和当前位 ===
        // 比如 sum 是 15。
        // carry 应该是 1  (用 Math.floor(sum / 10))
        // 当前位应该是 5  (用 sum % 10)
        carry = Math.floor(sum / 10);
        let digit = sum % 10;
     
        // === 第四步：造新节点挂到 l3 后面 ===
        current.next = new ListNode(digit);
     
        // === 第五步：移动指针 ===
        // current 必须往前走一步，站在刚造好的节点上，准备接下一个
        current = current.next;
     
        // l1 和 l2 也要往前走（前提是它们还没走到 null）
        if (l1 !== null) l1 = l1.next;
        if (l2 !== null) l2 = l2.next;
    }
 
    // 3. 返回假头后面的真链表
    return dummy.next;
};

/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function(x) {
    // 1. 负数直接排除，秒杀
    if (x < 0) return false;
 
    // 2. 转成字符串
    let str = x.toString();
 
    // 3. 双指针法（不用 slice，不占额外内存）
    let left = 0;
    let right = str.length - 1;
 
    while (left < right) {
        if (str[left] !== str[right]) {
            return false; // 只要有一对对不上，就不是回文
        }
        left++;
        right--;
    }
 
    return true; // 全都对上了
};

function isPrime(num) {
    if (num < 2) return false;
    // ⚠️ 优化重点：只要循环到 开根号(num) 就行了
    // 如果循环到 num，LeetCode 会超时 (Time Limit Exceeded)
    for (let i = 2; i <= Math.sqrt(num); i++) {
        if (num % i === 0) return false;
    }
    return true;
}
